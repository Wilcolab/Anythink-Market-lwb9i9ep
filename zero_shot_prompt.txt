def to_snake_case(text):
    import re
    text = re.sub(r'[\s\-]+', '_', text)
    text = re.sub(r'([a-z0-9])([A-Z])', r'\1_\2', text)
    return text.lower()
    def to_camel_case(text):
        import re
        text = re.sub(r'[\s\-_]+', ' ', text).title().replace(' ', '')
        return text[0].lower() + text[1:] if text else ''
